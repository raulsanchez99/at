3. Crear filas (tr) con celdas (td)

    const tr = document.createElement("tr"); // Nueva fila

    const td = document.createElement("td"); // Celda para imagen + nombre
    td.id = "producto_" + producto.id;

    // Imagen del producto
    const img = document.createElement("img");
    img.src = `images/${producto.id}.jpg`;
    img.alt = producto.nombre;
    img.style.width = "100px";

    td.appendChild(img);
    td.appendChild(document.createTextNode(producto.nombre));

    tr.appendChild(td);
    tbodyMenu.appendChild(tr);
});
